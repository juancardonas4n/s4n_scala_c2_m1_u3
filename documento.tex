\[
  recursive(m,n) =
  \begin{cases}
    m = n, & n \\
    m > n, & recursive(m-n,n) \\
    m \leq n & recursive(n-m,m) \\
  \end{cases}
\]


\[
  binomial(m,n) =
  \begin{cases}
    k = 0 \vee n = k & 1 \\
    \text{de otra forma} & binomial(n - 1,k - 1) + binomial(n - 1, k) \\
  \end{cases}
\]
